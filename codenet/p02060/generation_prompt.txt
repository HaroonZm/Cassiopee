```python
from math import ceil
N = int(input())
P = tuple(map(int, input().split()))
T = tuple(map(int, input().split()))
ans = 10**10
for i in range(1+ceil(N/T[0])):
    for j in range(1+ceil(N/T[1])):
        for k in range(1+ceil(N/T[2])):
            l = max(0, ceil((N-i*T[0]-j*T[1]-k*T[2])/T[3]))
            ans = min(ans, P[0]*i + P[1]*j + P[2]*k + P[3]*l)
print(ans)
```
