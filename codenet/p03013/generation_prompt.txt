```python
n, m = map(int, input().split())
a = [int(input()) for _ in range(m)]
dp = [0 for _ in range(n+10)]
dp[0] = 1
for x in a:
    dp[x] = -1
mod = 10**9 + 7
for i in range(n):
    if dp[i] < 0:
        continue
    if dp[i+1] >= 0:
        dp[i+1] += dp[i]
        dp[i+1] %= mod
    if dp[i+2] >= 0:
        dp[i+2] += dp[i]
        dp[i+2] %= mod
print(dp[n])
```
