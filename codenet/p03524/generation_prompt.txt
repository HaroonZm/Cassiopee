```python
s=input()
l=len(s)
cnt1=0
cnt2=0
cnt3=0
for i in range(l):
  if s[i]=='a':
    cnt1+=1
  elif s[i]=='b':
    cnt2+=1
  elif s[i]=='c':
    cnt3+=1
if max(cnt1,cnt2,cnt3)==1:
  print('YES')
elif cnt1==0 or cnt2==0 or cnt3==0:
  print('NO')
elif max(cnt1,cnt2,cnt3)-min(cnt1,cnt2,cnt3)<=1:
  print('YES')
else:
  print('NO')
```
