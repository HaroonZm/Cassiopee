import sys
from sys import stdin
from math import ceil
input = stdin.readline

def main(args):
    L = int(input())
    A = int(input())
    B = int(input())
    C = int(input())
    D = int(input())

    days_to_finish_A = ceil(A / C)
    days_to_finish_B = ceil(B / D)
    print(L - max(days_to_finish_A, days_to_finish_B))

if __name__ == '__main__':
    main(sys.argv[1:])
