```python
MOD = 10**9+7
n,m=map(int,input().split())
x=list(map(int,input().split()))
y=list(map(int,input().split()))
xx = 0
for i in range(n-1):
    w = x[i+1] - x[i]
    c = (i+1)*(n-i-1)
    xx = (xx + w * c) % MOD
yy = 0
for i in range(m-1):
    h = y[i+1] - y[i]
    c = (i+1)*(m-i-1)
    yy = (yy + h * c) % MOD
print((xx*yy) % MOD)
```
