```python
n,m = map(int,input().split())
lst = [0]
for i in range(n - 1):
  lst.append(lst[-1] + int(input()))
ans = 0
num = 0
for i in range(m):
  a = int(input())
  ans += abs(lst[num] - lst[num + a])
  ans %= 100000
  num += a
print(ans)
```
