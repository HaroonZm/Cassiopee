```python
N = int(input())
As = [A-1 for A in map(int, input().split())]
Bs = list(map(int, input().split()))
Cs = list(map(int, input().split()))

ans = 0
APrev = -100
for i in range(N):
    A = As[i]
    ans += Bs[A]
    if APrev == A-1:
        ans += Cs[APrev]
    APrev = A

print(ans)
```
