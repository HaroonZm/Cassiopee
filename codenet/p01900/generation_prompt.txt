n = int(input())
a = list(map(int, input().split()))
c = [0]*3
for i in a:	c[i%3] += 1
if (c[1]|c[2]) == 0: ans = 1
else:
	ans, n = c[0], n-c[0]
	if n <= 3: ans += n
	else:
		t = max(-3, min(3, c[1]-c[2]))
		if t > 0: ans += 2*c[2]+t
		else: ans += 2*c[1]-t
print(ans)
