```python
a,b=(int(x)-1 for x in input().split())
L=[]
for i in range(b+1):
    L.append(list(map(int,input().split())))
L.sort(key=lambda x:x[1]) 
ans=0
bridge_right=-1
#区間スケジューリング問題：終端でソート！（始端の場合も）
#終端が小さい方から見ていって、解決されてないときは一番右の橋を壊す←「一番右で損しない」が区間スケジューリングの性質
for l in L:
    if not l[0]<=bridge_right<l[1]:
        bridge_right=l[1]-1
        ans+=1
print(ans)
```
