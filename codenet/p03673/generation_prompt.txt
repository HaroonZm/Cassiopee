```python
from collections import deque

n = int(input())
a = list(map(int, input().split()))

if n % 2 == 0:
    flg = True
else:
    flg = False

na = deque()
for i in range(n):
    if i % 2 == 0:
        na.append(a[i])
    else:
        na.appendleft(a[i])

na = list(na)
if flg:
    for el in na:
        print(el, end=" ")
else:
    for el in reversed(na):
        print(el, end=" ")
```
