N, A = map(int, raw_input().split())
ans = 10 ** 18

for num in range(1, 65):
    p = int(N ** (1.0 / num))
    ls = [p] * num
    pos = 0
    while True:
        cnt = 1
        for p in ls:
            cnt *= p
        if cnt >= N:

        ls[pos] += 1

        pos %= num
