ans=[]
while True:
    scMax=0
    scMin=500
    n=input()
    if n==0:
        break
    for i in xrange(n):
        score=sum(map(int,raw_input().split()))
        if score > scMax:
            scMax=score
        if score < scMin:
            scMin=score
    ans.append(str(scMax)+' '+str(scMin))
    
for i in ans:
    print i
