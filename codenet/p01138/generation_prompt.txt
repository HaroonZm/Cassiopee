while True:
    n = int(raw_input())
    if n == 0:
        break
    l = []
    for _ in range(n):
        hms1, hms2 = raw_input().split()
        l.append((hms1, True))
        l.append((hms2, False))
    l.sort()
    r, t = 0, 0
    for i in range(len(l)):
        if l[i][1]:
            t += 1
        else:
            t -= 1
        r = max(r, t)
    print r
