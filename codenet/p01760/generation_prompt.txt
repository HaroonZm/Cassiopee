```python
import sys
T,D = [int(x) for x in sys.stdin.readline().split()]
t = [int(x) for x in sys.stdin.readline().split()]
d = [int(x) for x in sys.stdin.readline().split()]
ans = float("inf")
for i in range(D+1):
    va = d[0]*i
    if va > D:continue
    for j in range(D+1):
        vb = d[1]*j
        k = va+vb
        if k > D or k < 1:continue
        T_ = va*t[0]+vb*t[1]
        T_ /= k
        if abs(T-T_) < ans:
            ans = abs(T-T_)
print(ans)
```
