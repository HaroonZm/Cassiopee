while 1:
    n = int(input())
    if n == 0:
        break

    ports = []
    for _ in range(n):
        tmp = list(input())
        port = tmp[0]
        for i in range(len(tmp)):
            if i != len(tmp) - 1 and tmp[i] in list("aiueo"):
                port += tmp[i+1]
        ports.append(port)

    for i in range(1, 51):
        ext = []
        for p in ports:
            tmp = p[:i]
            if tmp in ext:
                break
            else:
                ext.append(tmp)
        else:
            print(i)
            break
    else:
        print(-1)
