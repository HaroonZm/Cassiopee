```python
n = int(input())
res = []
su = 0
for i in range(1, n+1):
    #print(res, i)
    res.append(i)
    if su + i > n:
        break
    su += i
sa = sum(res) - n
if sa == 0:
    print("\n".join(map(str, res)))
else:
    del res[res.index(sa)]
    print("\n".join(map(str, res)))
```
