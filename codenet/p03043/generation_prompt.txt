```python
from math import log2, ceil

N,K = map(int, input().split())

if N <= K:
	lis = [0]*N
	cnt = N + 1
	score = 0
else:
	lis = [0]*(K-1)
	cnt = K
	score = (N-K+1)/N

for i in range(1, cnt):
	lis[i-1] = (1/N) * pow(1/2, ceil(log2(K/i)))
print(sum(lis)+score)
```
