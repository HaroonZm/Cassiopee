import collections

class CursorList():
    def __init__(self):
        self.vector = collections.deque()
        self.cursor = 0

    def insert(self, x):
        self.vector.appendleft(x)
        return self

    def move(self, d):
        self.vector.rotate(-d)
        self.cursor += d
        return self

    def erase(self):
        self.vector.popleft()
        return self

num_queue = int((input()))
L = CursorList()

for i in range(0, num_queue):
    queue = tuple(map(int, input().split(' ')))
    if queue[0] == 0:
        L.insert(queue[1])
    elif queue[0] == 1:
        L.move(queue[1])
    elif queue[0] == 2:
        L.erase()

L.vector.rotate(L.cursor)
print('\n'.join(map(str, L.vector)))
