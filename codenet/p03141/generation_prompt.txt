```python
import numpy as np
N = int(input())
A = [0]*N
B = [0]*N
for i in range(N):
    A[i], B[i] = map(int, input().split())

a = np.array(A)
b = np.array(B)
c = np.zeros(N)

# cはaとbの要素をすべて足したもの
c = a + b
# 降順にソート
csorted = np.sort(c)[::-1]
# bの総和
bsum = np.sum(b)
# csortedの奇数番目を足していって最後にbsumの値を引けば求まる。
csum = 0
for i in range(N):
    if i % 2 == 0:
        csum += int(csorted[i])

ans = csum - bsum

print(ans)
```
