```python
A, B = map(int, input().split())

def prime_decompose(x):
    decomposed = []
    for i in range(2,int(pow(x,0.5))+1):
        while x % i == 0:
            x //= i
            decomposed.append(i)
    if x > 1:
        decomposed.append(x)
    return decomposed

A_dec = set(prime_decompose(A))
B_dec = set(prime_decompose(B))

ans = 1
for a in A_dec:
    if a in B_dec:
        ans += 1

print(ans)
```
