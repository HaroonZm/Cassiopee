    min_list=(sorted(num,reverse=True))
    #print(min_list)
    max_list=sorted(num)
    #print(max_list)
    max_num=0
    min_num=0
    for i in range(len(num)):
        max_num+=max_list[i]*(10**i)
        min_num+=min_list[i]*(10**i)
    return max_num-min_num
