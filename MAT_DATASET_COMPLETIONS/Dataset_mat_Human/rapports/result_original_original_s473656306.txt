    b %= a
    return b if a%b == 0 else GCD(b, a)

    try:
        a, b = list(map(float, input().split()))
    except EOFError:
        break
    if a >= b:
        a, b = b, a

    print("{0} {1}".format(int(gcd), int(a*b/gcd)))
