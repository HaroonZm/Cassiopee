# BFSで伐採工場(0,0)から各場所への最短距離を求める
from collections import deque

def bfs():
    dist = [[-1]*W for _ in range(H)]
    dist[0][0] = 0
    queue = deque()
    queue.append((0,0))
    while queue:
        x, y = queue.popleft()
        for dx, dy in [(1,0),(-1,0),(0,1),(0,-1)]:
            nx, ny = x+dx, y+dy
            if 0 <= nx < H and 0 <= ny < W:
                if A[nx][ny] == 0 and dist[nx][ny] == -1:
                    dist[nx][ny] = dist[x][y] + 1
                    queue.append((nx, ny))
    return dist

dist = bfs()
