import sys
import math

def polygon_area(angles):
    angles.append(360 - sum(angles))
    coords = [(1, 0)]
    angle_sum = 0
    for v in angles:
        angle_sum += v
        rad = math.radians(angle_sum)
        coords.append((math.cos(rad), math.sin(rad)))
    # Shoelace formula
    area = 0
    for i in range(len(coords)):
        x1, y1 = coords[i]
        x2, y2 = coords[(i+1) % len(coords)]
        area += x1 * y2 - y1 * x2
    return abs(area) / 2
