def bfs(start, goal):

    queue = deque()
    queue.append(start)
    dist = [[-1]*W for _ in range(H)]
    dist[start[0]][start[1]] = 0

    while queue:

        if (x, y) == goal:

                    dist[nx][ny] = dist[x][y] + 1
                    queue.append((nx, ny))
    return -1  # 不可能な場合
