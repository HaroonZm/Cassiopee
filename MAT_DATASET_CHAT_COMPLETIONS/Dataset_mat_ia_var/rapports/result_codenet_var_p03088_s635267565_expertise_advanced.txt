def is_forbidden(x, y, z, w):
    triplets = [
        (z, y, x),
        (z, w, x),
        (w, y, x)
    ]
    return any(t in forbidden for t in triplets)

@lru_cache(maxsize=None)
def dp(pos, x, y, z):
    if pos == N:
        return 1
    total = 0

    return total
